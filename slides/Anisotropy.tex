  \begin{itemize}
  \item Anisotropy
    \begin{itemize}

    \item Line smoothers

    \begin{itemize}
    \item Make coarse grids align
    \item Strong smoothers
    \item Energy-minimizing interpolants
    \end{itemize}
  \end{itemize}
\end{frame}
