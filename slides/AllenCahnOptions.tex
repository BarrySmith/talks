\[ \bm{u}(x,t) \in G = \{\bm{v}\in \mathbb{R}^{d}| v_i \geq 0, \sum_{i = 1}^{N} v_i = 1\},\quad \forall (x,t) \in Q. \]

-I & -I & -I & 0 
\end{pmatrix}, \qquad
P = 
\begin{pmatrix}
A  & 0  & 0  & 0 \\
0  & A  & 0  & 0 \\
0  & 0  & A  & 0 \\
-I & -I & -I & S_{\text{LSC}}
\end{pmatrix}
\end{equation*}
